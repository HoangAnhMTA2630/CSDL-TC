WITH ma_doc_with_nam AS (
  SELECT 
    *,
    CASE 
      WHEN ngay_tiep_nhan IS NOT NULL THEN EXTRACT(YEAR FROM ngay_tiep_nhan)::text
      ELSE 'Không rõ'
    END AS nam
  FROM public.ma_doc
)
SELECT 
  nam,
  CASE 
    WHEN bao_cao_id IS NULL THEN 'Không có báo cáo'
    ELSE 'Có báo cáo'
  END AS co_bao_cao,
  COUNT(*) AS so_luong
FROM ma_doc_with_nam
GROUP BY nam, 
         CASE 
           WHEN bao_cao_id IS NULL THEN 'Không có báo cáo'
           ELSE 'Có báo cáo'
         END
ORDER BY 
  CASE 
    WHEN nam ~ '^\d+$' THEN CAST(nam AS INT)
    ELSE 9999
  END;
